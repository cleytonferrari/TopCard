DROP VIEW VIPAVO;

  "VIPAVO_ID",
  "NOME",
  "DATANASCIMENTO",
  "PAI",
  "MAE",
  "FORMADEPAGAMENTO",
  "CC",
  "AG",
  "STATUS",
  "CPF",
  "RG",
  "DATACADASTRO",
  "TELEFONES",
  "EMAIL",
  "OBS",
  "ESTADOCIVIL",
  "PROFISSAO",
  "NACIONALIDADE",
  "SEXO",
  "ENDERECOS_ID",
  "BANCO_ID",
  "NATURAL_ID",
  "VIPPAI_ID",
  "NUMERO",
  "CONTRATO"
) AS

 SELECT
CLIENTE_ID, NOME, DATANASCIMENTO, PAI, MAE, FORMADEPAGAMENTO, CC, AG, STATUS, CPF, RG, DATACADASTRO, TELEFONES, EMAIL, OBS,
   ESTADOCIVIL, PROFISSAO, NACIONALIDADE, SEXO, ENDERECOS_ID, BANCO_ID, NATURAL_ID, VIPPAI_ID, NUMERO, CONTRATO
FROM
   CLIENTE
;

DROP VIEW VIPPAI;

CREATE VIEW "VIPPAI" (
  "VIPPAI_ID",
  "NOME",
  "DATANASCIMENTO",
  "PAI",
  "MAE",
  "FORMADEPAGAMENTO",
  "CC",
  "AG",
  "STATUS",
  "CPF",
  "RG",
  "DATACADASTRO",
  "TELEFONES",
  "EMAIL",
  "OBS",
  "ESTADOCIVIL",
  "PROFISSAO",
  "NACIONALIDADE",
  "SEXO",
  "NUMERO",
  "CONTRATO",
  "ENDERECOS_ID",
  "BANCO_ID",
  "NATURAL_ID",
  "VIPAVO_ID",
  "PAINOME",
  "PAICPF"
) AS

 SELECT
   CLIENTE.CLIENTE_ID, CLIENTE.NOME, CLIENTE.DATANASCIMENTO, CLIENTE.PAI, CLIENTE.MAE, CLIENTE.FORMADEPAGAMENTO,
   CLIENTE.CC, CLIENTE.AG, CLIENTE.STATUS, CLIENTE.CPF, CLIENTE.RG, CLIENTE.DATACADASTRO, CLIENTE.TELEFONES, CLIENTE.EMAIL,
   CLIENTE.OBS, CLIENTE.ESTADOCIVIL, CLIENTE.PROFISSAO, CLIENTE.NACIONALIDADE, CLIENTE.SEXO, CLIENTE.NUMERO, CLIENTE.CONTRATO, CLIENTE.ENDERECOS_ID,
   CLIENTE.BANCO_ID, CLIENTE.NATURAL_ID, CLIENTE.VIPPAI_ID, VIPAVO.NOME, VIPAVO.CPF
FROM
   CLIENTE,
   VIPAVO
WHERE
   CLIENTE.VIPPAI_ID =VIPAVO.VIPAVO_ID
;
